In [201]:
sepal_length_stand = (sepal_length - np.mean(sepal_length))/np.std(sepal_length)
petal_width_stand = (petal_width - np.mean(petal_width))/np.std(petal_width)
In [202]:
sepal_width_norm = (sepal_width - min(sepal_width))/(max(sepal_width) - min(sepal_width))
petal_length_norm = (petal_length - min(petal_length))/(max(petal_length) - min(petal_length))
